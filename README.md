install all libraries needed
```bash
pip install -r requirements.txt
```

```sql
create database wareHouseMS
```

```env
DB_URL="mssql+pyodbc://127.0.0.1/wareHouseMS?driver=ODBC+Driver+17+for+SQL+Server"
```
